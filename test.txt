from flask import Flask, render_template, jsonify
import json

app = Flask(__name__)

@app.route('/')
def home():
    return 'This is the home route.'

@app.route('/graph')
def graph():
    # Load data from the JSON file
    with open('history.json') as f:
        data = json.load(f)
    
    return render_template('graph.html', data=data)

@app.route('/data')
def get_data():
    with open('history.json') as f:
        data = json.load(f)
    return jsonify(data)

if __name__ == '__main__':
    app.run()
