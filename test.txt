import pickle

# Example variables
uid = 'my_username'
password = 'my_password'

# Pickling the variables separately
with open('uid.pickle', 'wb') as f:
    pickle.dump(uid, f)

with open('password.pickle', 'wb') as f:
    pickle.dump(password, f)

# Unpickling the variables
with open('uid.pickle', 'rb') as f:
    unpickled_uid = pickle.load(f)

with open('password.pickle', 'rb') as f:
    unpickled_password = pickle.load(f)

# Printing the variables
print("UID:", unpickled_uid)
print("Password:", unpickled_password)
