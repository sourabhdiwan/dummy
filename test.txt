import pickle

# Example object to be pickled
data = {'name': 'John', 'age': 30, 'city': 'New York'}

# Pickling the object
with open('data.pickle', 'wb') as f:
    pickle.dump(data, f)

# Unpickling the object
with open('data.pickle', 'rb') as f:
    unpickled_data = pickle.load(f)

print(unpickled_data)
