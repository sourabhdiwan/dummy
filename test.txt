import pandas as pd

def extract_field_from_feed(field_name):
    # Step 1: Read the flat text file 'feed'
    with open('feed.txt', 'r') as feed_file:
        feed_content = feed_file.read()

    # Step 2: Read the mapping file as an Excel sheet
    mapping_file_path = f'{field_name}_mapping.xlsx'
    mapping_data = pd.read_excel(mapping_file_path)

    # Step 3: Find the start position, end position, and length from the mapping data
    mapping_row = mapping_data[mapping_data['Field Name'] == field_name].iloc[0]
    start_pos = mapping_row['Start Position']
    end_pos = mapping_row['End Position']
    field_length = mapping_row['Length']

    # Step 4: Extract the value from the feed file based on the positions
    start_pos = int(start_pos)
    end_pos = int(end_pos)
    field_value = feed_content[start_pos - 1:end_pos].strip()

    return field_value, field_name
